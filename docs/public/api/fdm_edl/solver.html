 79        if cls is BaseSolver:

158    ) -> jax.Array:
159        """Clamp Dirichlet nodes in the solution vector."""
160        for bc in bcs:
161            if bc.is_dirichlet:

167        _phi, _grad, _src, _res, converged, n_iter = state
168        # continue iterating if not converged and under max iterations
169        return jnp.logical_and(~converged, n_iter < max_iter)

172    def _convergence_flag(a, b, atol: float, rtol: float) -> jax.Array:
173        if atol is not None or rtol is not None:
174            _atol = atol if atol is not None else 1e-8
175            _rtol = rtol if rtol is not None else 1e-5
176            return jnp.allclose(a, b, atol=_atol, rtol=_rtol)
177        else:
178            return jnp.bool_(True)

 80        self,
 81        residual_fn: ResidualFunction,
 82        phi: jax.Array,
 83        boundary_conditions: Sequence[BoundaryCondition],
 84        *args,
 85    ) -> RootSolveResult:
 86        """
 87        Solve via Newton-Raphson iterations with a full JAX Jacobian.
 88
 89        The iteration loop uses ``jax.lax.while_loop`` so the entire
 90        solve is JIT-compilable.  ``while_loop`` is preferred over
 91        ``fori_loop`` because it supports early termination on
 92        convergence, avoiding unnecessary Jacobian evaluations.
 93
 94        Parameters
 95        ----------
 96        residual_fn : callable
 97            Residual function with signature
 98            ``residual_fn(phi, *args) -> Tuple[jax.Array, Tuple[jax.Array, jax.Array, jax.Array]]``.
 99        phi : jax.Array
100            Initial guess.
101        boundary_conditions : Sequence[BoundaryCondition]
102            List of boundary conditions.
103        *args
104            Extra positional arguments forwarded to *residual_fn*.
105
106        Returns
107        -------
108        RootSolveResult
109            Result containing the solution and solver diagnostics.
110        """
111        # clamp Dirichlet nodes in the initial guess
112        phi_init = self._clamp_dirichlet_nodes(phi, boundary_conditions)
113        res_init, _aux = residual_fn(phi_init, boundary_conditions, *args)
114        grad_init, _div_D, src_init = _aux
115
116        # Inner JIT-compiled solve.  residual_fn, boundary_conditions,
117        # and extra args are captured by closure (compile-time constants);
118        # only `phi` is a traced dynamic argument.
119        @jax.jit
120        def _solve(phi):
121            def cond_fn(state):
122                return self._cond_fn(state, self.max_iter)
123
124            def body_fn(state):
125                phi_old, grad_old, src_old, res_old, _, n_iter = state
126
127                jac, _aux = jax.jacobian(residual_fn, has_aux=True)(
128                    phi_old, boundary_conditions, *args
129                )
130                step = jnp.linalg.solve(jac, -res_old).reshape(phi_old.shape)
131
132                # Backtracking line search to reduce divergence/non-finite updates.
133                def ls_body(_ii, ls_state):
134                    alpha, best_phi, best_norm, accepted = ls_state
135                    cand_phi = self._clamp_dirichlet_nodes(
136                        phi + alpha * step, boundary_conditions
137                    )
138                    cand_res, _aux = residual_fn(cand_phi, boundary_conditions, *args)
139                    cand_norm = self._residual_norm(cand_res)
140                    cand_finite = jnp.logical_and(
141                        jnp.all(jnp.isfinite(cand_phi)),
142                        jnp.all(jnp.isfinite(cand_res)),
143                    )
144                    improved = jnp.logical_and(cand_finite, cand_norm < best_norm)
145                    accepted = jnp.logical_or(accepted, improved)
146                    best_phi = jnp.where(improved, cand_phi, best_phi)
147                    best_norm = jnp.where(improved, cand_norm, best_norm)
148                    return (0.5 * alpha, best_phi, best_norm, accepted)
149
150                # alpha, best_phi, best_norm, accepted
151                ls_init = (
152                    jnp.asarray(self.alpha, dtype=phi.dtype),
153                    phi_old,
154                    self._residual_norm(res_old),
155                    jnp.bool_(False),
156                )
157                _, ls_phi, _ls_norm, accepted = jax.lax.fori_loop(
158                    0, self.max_iter_ls, ls_body, ls_init
159                )
160
161                full_phi = self._clamp_dirichlet_nodes(
162                    phi_old + step, boundary_conditions
163                )
164                phi_new = jnp.where(accepted, ls_phi, full_phi)
165                res_new, _aux = residual_fn(phi_new, boundary_conditions, *args)
166                grad_new, _div_D, src_new = _aux
167
168                # check convergence on step, gradient, source
169                converged_step = self._convergence_flag(
170                    phi_new, phi_old, self.atol_var, self.rtol_var
171                )
172                converged_grad = self._convergence_flag(
173                    grad_new, grad_old, self.atol_grad, self.rtol_grad
174                )
175                converged_src = self._convergence_flag(
176                    src_new, src_old, self.atol_src, self.rtol_src
177                )
178                # check the residual
179                converged_res = self._convergence_flag(
180                    res_new, jnp.zeros_like(res_new), self.atol_res, self.rtol_res
181                )
182
183                converged = (
184                    converged_step * converged_grad * converged_src * converged_res
185                )
186                return (phi_new, grad_new, src_new, res_new, converged, n_iter + 1)
187
188            init_state = (
189                phi_init,
190                grad_init,
191                src_init,
192                res_init,
193                jnp.bool_(False),
194                jnp.int32(0),
195            )
196            phi_final, grad_final, src_final, res_final, converged, n_iter = (
197                jax.lax.while_loop(cond_fn, body_fn, init_state)
198            )
199            # clamp once more after the final step update
200            phi_final = self._clamp_dirichlet_nodes(phi_final, boundary_conditions)
201            # Cut gradients: differentiating through the Newton loop is
202            # not useful and would checkpoint every iteration's Jacobian.
203            phi_final = jax.lax.stop_gradient(phi_final)
204            return phi_final, grad_final, src_final, res_final, converged, n_iter
205
206        phi, grad, src, res, converged, n_iter = _solve(phi)
207        return RootSolveResult(
208            solution=phi,
209            converged=converged,
210            n_iter=n_iter,
211            residual=res,
212            gradient=grad,
213            source=src,
214        )
