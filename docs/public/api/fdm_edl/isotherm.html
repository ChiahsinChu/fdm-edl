143    def _compute(
144        self,
145        _phi: float,
146        ignore_convergence_failure: bool,
147        *,
148        max_iter: int = 500,
149        lr: float = 0.05,
150        tol: float = 1e-10,
151        x0: float = 0.0,
152    ) -> float | jax.Array:
153        """Solve the isotherm for a single scalar potential value.
154
155        Uses a sigmoid reparametrisation to keep :math:`\\theta` in
156        :math:`(0, \\theta_{\\max})` and minimises the squared residual
157        with an Adam optimiser inside a ``jax.lax.while_loop``.
158
159        Parameters
160        ----------
161        _phi : float
162            Electrode potential in volts (dimensionless scalar).
163        max_iter : int, optional
164            Maximum optimiser iterations.
165        lr : float, optional
166            Adam learning rate.
167        tol : float, optional
168            Squared-residual convergence tolerance.
169        x0 : float, optional
170            Initial unconstrained variable value.
171
172        Returns
173        -------
174        float or jax.Array
175            Equilibrium surface coverage :math:`\\theta`.
176        """
177        residual = self._residual_fn(_phi)
178
179        eps = 1e-12
180
181        def x_to_theta(x):
182            # theta in (0, theta_max), avoid hitting endpoints exactly
183            s = jax.nn.sigmoid(x)
184            return eps + (self.theta_max - 2 * eps) * s
185
186        def loss_fn(x):
187            th = x_to_theta(x)
188            r = residual(th)
189            return r * r
190
191        opt = optax.adam(lr)
192        opt_state = opt.init(x0)
193
194        x = jnp.asarray(x0)
195
196        grad_fn = jax.grad(loss_fn)
197
198        def body(state):
199            x, opt_state, _prev_loss, i = state
200            g = grad_fn(x)
201            updates, opt_state = opt.update(g, opt_state, x)
202            x = optax.apply_updates(x, updates)
203            cur_loss = loss_fn(x)
204            return x, opt_state, cur_loss, i + 1
205
206        def cond(state):
207            x, opt_state, cur_loss, i = state
208            return jnp.logical_and(cur_loss > tol, i < max_iter)
209
210        init_loss = loss_fn(x)
211        x, opt_state, final_loss, _ = jax.lax.while_loop(
212            cond, body, (x, opt_state, init_loss, 0)
213        )
214        if (cond((x, opt_state, final_loss, -1))) and (not ignore_convergence_failure):
215            raise RuntimeError(f"Final residual: {jnp.sqrt(jnp.mean(final_loss ** 2))}")
216        # print(_)
217        # print(f"Final residual: {jnp.sqrt(jnp.mean(final_loss ** 2))}")
218
219        theta = x_to_theta(x)
220        return theta
221
222    def _residual_fn(self, phi: float) -> Callable[[float], float | jax.Array]:
223        """Build the isotherm residual function for a given potential.
224
225        The residual is
226
227        .. math::
228
229            r(\\theta) = \\ln\\!\\left(\\frac{\\theta}{\\theta_{\\max} - \\theta}\\right)
230            + g(\\theta) - \\frac{n_{\\mathrm{et}} F}{RT}\\,\\phi
231
232        where :math:`g(\\theta)` is the lateral-interaction term.
233
234        Parameters
235        ----------
236        phi : float
237            Electrode potential in volts (dimensionless scalar).
238
239        Returns
240        -------
241        Callable[[float], float | jax.Array]
242            A function mapping coverage :math:`\\theta` to the residual.
243        """
244        rhs_coeff = (
245            (
246                (self.n_et * constants.FARADAY_CONSTANT)
247                / (constants.MOLAR_GAS_CONSTANT * self.temperature)
248            )
249            .to("1/V")
250            .value
251        )
252
253        # Choose eps relative to theta_max scale; can be a parameter if you want.
254        eps = 1e-12
255
256        def residual_fn(theta: float) -> float | jax.Array:
257            theta = jnp.asarray(theta)
258
259            # Work in normalized coverage y in (0, 1)
260            y = theta / self.theta_max
261            y = jnp.clip(y, eps, 1.0 - eps)
262
263            # stable log(y/(1-y)) = log(y) - log1p(-y)
264            log_term = jnp.log(y) - jnp.log1p(-y)
265
266            lhs = log_term + self.lateral_interaction(theta)
267            rhs = -rhs_coeff * phi
268            return lhs - rhs
269
270        return residual_fn

Frumkin isotherm with coverage-dependent lateral interactions.

The dimensionless lateral-interaction term is

$$g(\theta) = \frac{\omega}{RT}\,(\alpha\,\theta + \beta)\,\theta$$

where \( \omega \) is the lateral-interaction energy, \( \alpha \) and \( \beta \) are shape parameters.
