 32    equation, solved with a finite-difference method.
 33
 34    Supports 1-D, 2-D, and 3-D grids.  Coordinates are expected as
 35    ``(n_grid, n_dim)``; 1-D arrays ``(n_grid,)`` are accepted and
 36    reshaped automatically.
 37
 38    Parameters
 39    ----------
 40    params : dict or str or pathlib.Path
 41        Model parameters as a dictionary or as a path to a JSON/YAML file.
 42        Required keys:
 43
 44        ``temperature`` : float
 45            Absolute temperature in Kelvin.
 46
 47        Optional keys:
 48
 49        ``epsilon_r`` : float
 50            Relative permittivity of the solvent (default: 78.5).
 51        ``electrode`` : dict
 52            Keyword arguments forwarded to :class:`~fdm_edl.edl.Electrode`.
 53        ``electrolyte`` : dict
 54            Keyword arguments forwarded to
 55            :class:`~fdm_edl.edl.Electrolyte`.
 56        ``solver`` : dict
 57            Keyword arguments forwarded to :class:`~fdm_edl.solver.BaseSolver`.
 58
 59    Attributes
 60    ----------
 61    temperature : float
 62        System temperature.
 63    dim : int
 64        Spatial dimension of the model (default: 1).
 65    electrode : Electrode
 66        Electrode component of the EDL model.
 67    electrolyte : Electrolyte
 68        Electrolyte component of the EDL model.
 69    solver : BaseSolver
 70        Nonlinear root-finding solver.
 71    result : RootSolveResult or None

270    def compute(
271        self,
272        coordinates: unxt.Quantity,
273        boundary_conditions: Sequence[BoundaryCondition],
274        phi0: unxt.Quantity | None = None,
275        ignore_convergence_failure: bool = False,
276    ):
277        """
278        Solve for the electrostatic potential profile.
279
280        Parameters
281        ----------
282        coordinates : unxt.Quantity, shape (n_grid,) or (n_grid, n_dim)
283            Grid coordinates.  1-D arrays are accepted and reshaped to
284            ``(n_grid, 1)`` internally.
285        boundary_conditions : list of BoundaryCondition
286            A list of :class:`~fdm_edl.bc.BoundaryCondition` objects that
287            define the constraints on specific nodes.
288        phi0 : unxt.Quantity or None, optional
289            Initial guess for the potential at all grid nodes, shape
290            ``(n_grid,)``.  When ``None`` (default), a zero array is
291            created and seeded with boundary-condition values.
292
293        Returns
294        -------
295        None
296            The result is stored in ``self.result``.
297        """
298        # convert coordinates to internal unit system
299        # Angstrom
300        _coordinates = coordinates.to(uc.UNIT_SYSTEMS["metal"]["length"]).value
301        # Volt
302        _phi0 = (
303            phi0.to(uc.UNIT_SYSTEMS["metal"]["electrical potential"]).value
304            if phi0 is not None
305            else None
306        )
307
308        # --- Normalise coordinates to (n_grid, n_dim) -----------------------
309        if _coordinates.ndim == 1:
310            coordinates_2d = _coordinates.reshape(-1, 1)
311        else:
312            coordinates_2d = _coordinates
313        n_grid = coordinates_2d.shape[0]
314        assert (
315            coordinates_2d.shape[1] == self.dim
316        ), "Coordinate dimension does not match model dimension."
317
318        if self.dim != 1:
319            raise NotImplementedError("Only dim=1 is currently supported.")
320
321        # For 1-D, use the flat coordinate vector
322        coordinates_1d = coordinates_2d[:, 0]
323
324        # --- Zero initial guess if not provided ----------------
325        if _phi0 is None:
326            _phi0 = jnp.zeros((n_grid,))
327        _phi0 = jnp.asarray(_phi0)
328
329        # --- Solve -----------------------------------------------------------
330        self._solver_result = self.solver.solve(
331            self.compute_residual,
332            _phi0,
333            boundary_conditions,
334            coordinates_1d,
335        )
336        assert self._solver_result is not None
337        if (not self._solver_result.converged) and (not ignore_convergence_failure):
338            raise RuntimeError(
339                f"Solver failed to converge after {self._solver_result.n_iter} iterations. "
340                f"Final residual: {jnp.sqrt(jnp.mean(self._solver_result.residual ** 2))}"
341            )
342
343        #  --- post-process results ------------------------------------------------
344        phi_final = self._solver_result.solution
345        grad, _ = self.grad_op(
346            coordinates_1d,
347            phi_final,
348        )
349        efield = cast(
350            unxt.Quantity,
351            unxt.Quantity(
352                -grad, unit=uc.UNIT_SYSTEMS["metal"]["electrical field strength"]
353            ).to(uc.UNIT_SYSTEMS[self._unit_system]["electrical field strength"]),
354        )
355        ion_conc = self.charge_model.ion_concentration_profile(
356            unxt.Quantity(
357                phi_final,
358                unit=uc.UNIT_SYSTEMS["metal"]["electrical potential"],
359            ),
360            self.electrolyte,
361            self.temperature,
362        )
363        rho_ion = charge_density_profile(ion_conc)
364        # calculate sigma from E-field at the electrode surface
365        # use eps(E_OHP) rather than eps_0!
366        eps_r = self.electrolyte.solvent.compute_eps(efield)
367        self.result = EDLStatus(
368            coordinate=coordinates,
369            sigma=cast(
370                unxt.Quantity,
371                (eps_r[0] * constants.VACUUM_PERMITTIVITY * efield[0]).to(
372                    uc.UNIT_SYSTEMS[self._unit_system]["surface charge density"]
373                ),
374            ),
375            phi=cast(
376                unxt.Quantity,
377                unxt.Quantity(
378                    phi_final,
379                    unit=uc.UNIT_SYSTEMS["metal"]["electrical potential"],
380                ).to(uc.UNIT_SYSTEMS[self._unit_system]["electrical potential"]),
381            ),
382            efield=efield,
383            rho=cast(
384                unxt.Quantity,
385                rho_ion.to(
386                    uc.UNIT_SYSTEMS[self._unit_system]["electrical charge density"]
387                ),
388            ),
389            ion_conc=ion_conc,
390            epsilon_r=eps_r,
391        )

393    def compute_residual(
394        self,
395        phi: jax.Array,
396        boundary_conditions: Sequence[BoundaryCondition],
397        coordinates: jax.Array,
398    ) -> Tuple[jax.Array, Tuple[jax.Array, jax.Array, jax.Array]]:
399        """
400        Compute the Poisson-Boltzmann residual at all grid nodes.
401
402        The physics equation ∇²φ + ρ_ion/ε = 0 is evaluated everywhere,
403        then each boundary condition overwrites its nodes' residual entries
404        with the appropriate constraint.
405        Note: Use jax.Array rather than unxt.Quantity for the input and output of this function
406
407        Parameters
408        ----------
409        phi : jax.Array, shape (n_grid,)
410            Electrostatic potential at every grid node.
411        boundary_conditions : sequence of BoundaryCondition
412            Boundary conditions to enforce.
413        coordinates : jax.Array, shape (n_grid,) or (n_grid, n_dim)
414            Grid coordinates (2-D array).
415
416        Returns
417        -------
418        residual : jax.Array, shape (n_grid,)
419            Residual at each node.  Zero when the discretised
420            Poisson-Boltzmann equation and all BCs are satisfied.
421        """
422
423        # --- Gradient of phi and divergence of electric displacement ---------
424        g_phi, div_D = self.grad_op(coordinates, phi)
425
426        # --- Ionic charge density (delegated to charge model) ----------------
427        rho_ion = self.charge_model.charge_density(
428            jnp.asarray(phi), self.electrolyte, self._temperature
429        )
430
431        eps_0 = constants.VACUUM_PERMITTIVITY.to(
432            uc.UNIT_SYSTEMS["metal"]["permittivity"]
433        ).value
434        # --- Physics residual at all nodes -----------------------------------
435        # residual: e/Angstrom^3
436        residual = div_D * eps_0 - rho_ion
437
438        # --- Apply boundary conditions -----------
439        for bc in boundary_conditions:
440            bc.update_coefficients(
441                eps_r=self.electrolyte.solvent._compute_eps(jnp.abs(g_phi[0])),
442            )
443            residual = bc.update_residual(residual, phi, g_phi)
444
445        return residual, (g_phi, div_D, rho_ion)

447    def _check_grad_op(self, grad_op: BaseGradientOP | None) -> BaseGradientOP:
448        """
449        Ensure a gradient operator exists and return a jitted callable.

451        If `grad_op` is None, we construct one based on (dim, solvent.type).
452        """

460    def _validate_user_grad_op(self, grad_op: BaseGradientOP) -> None:
461        solvent_type = self.electrolyte.solvent.type
462
463        # EuclideanFDOp and the derived class AxisymmetricFDOp are not allowed for non-uniform solvents
464        if solvent_type != "uniform" and ("FDOp" in type(grad_op).__name__):
465            raise ValueError(
466                f"Invalid grad_op for solvent type '{solvent_type}': finite_difference is not allowed. "
467                "Use finite_volume (or pass a different BaseGradientOP)."
468            )

470    def _build_default_grad_op(self) -> BaseGradientOP:
471        """Construct the default gradient operator for the current configuration."""
472        if self.dim != 1:
473            raise NotImplementedError(
474                f"Numerical gradient operator for dim={self.dim} is not yet implemented. "
475                "Pass a BaseGradientOP explicitly."
476            )
477
478        solvent_type = self.electrolyte.solvent.type
479        grad_op_class = self._default_grad_op_class_for_solvent(solvent_type)
480
481        return grad_op_class(eps_func=self.electrolyte.solvent._compute_eps)
482
483    @staticmethod
484    def _default_grad_op_class_for_solvent(solvent_type: str) -> type[BaseGradientOP]:
485        """Map solvent type -> gradient operator class."""
486        mapping: dict[str, type[BaseGradientOP]] = {
487            "uniform": EuclideanFDOp,
488            "langevin": EuclideanFVOp,
489            "booth": EuclideanFVOp,
490        }
491        try:
492            return mapping[solvent_type]
493        except KeyError as e:
494            raise NotImplementedError(
495                f"Automatic grad_op construction for solvent type '{solvent_type}' is not implemented. "
496                "Pass a BaseGradientOP explicitly."
497            ) from e

Supports 1-D, 2-D, and 3-D grids. Coordinates are expected as (n_grid, n_dim); 1-D arrays (n_grid,) are accepted and reshaped automatically.

  • ValueError: If params is not a dict or a valid path, or if temperature is missing from the parameter dictionary.

def compute_residual( self, phi: jax.Array, boundary_conditions: Sequence[fdm_edl.utils.bc.BoundaryCondition], coordinates: jax.Array) -> Tuple[jax.Array, Tuple[jax.Array, jax.Array, jax.Array]]: 
393    def compute_residual(
394        self,
395        phi: jax.Array,
396        boundary_conditions: Sequence[BoundaryCondition],
397        coordinates: jax.Array,
398    ) -> Tuple[jax.Array, Tuple[jax.Array, jax.Array, jax.Array]]:
399        """
400        Compute the Poisson-Boltzmann residual at all grid nodes.
401
402        The physics equation ∇²φ + ρ_ion/ε = 0 is evaluated everywhere,
403        then each boundary condition overwrites its nodes' residual entries
404        with the appropriate constraint.
405        Note: Use jax.Array rather than unxt.Quantity for the input and output of this function
406
407        Parameters
408        ----------
409        phi : jax.Array, shape (n_grid,)
410            Electrostatic potential at every grid node.
411        boundary_conditions : sequence of BoundaryCondition
412            Boundary conditions to enforce.
413        coordinates : jax.Array, shape (n_grid,) or (n_grid, n_dim)
414            Grid coordinates (2-D array).
415
416        Returns
417        -------
418        residual : jax.Array, shape (n_grid,)
419            Residual at each node.  Zero when the discretised
420            Poisson-Boltzmann equation and all BCs are satisfied.
421        """
422
423        # --- Gradient of phi and divergence of electric displacement ---------
424        g_phi, div_D = self.grad_op(coordinates, phi)
425
426        # --- Ionic charge density (delegated to charge model) ----------------
427        rho_ion = self.charge_model.charge_density(
428            jnp.asarray(phi), self.electrolyte, self._temperature
429        )
430
431        eps_0 = constants.VACUUM_PERMITTIVITY.to(
432            uc.UNIT_SYSTEMS["metal"]["permittivity"]
433        ).value
434        # --- Physics residual at all nodes -----------------------------------
435        # residual: e/Angstrom^3
436        residual = div_D * eps_0 - rho_ion
437
438        # --- Apply boundary conditions -----------
439        for bc in boundary_conditions:
440            bc.update_coefficients(
441                eps_r=self.electrolyte.solvent._compute_eps(jnp.abs(g_phi[0])),
442            )
443            residual = bc.update_residual(residual, phi, g_phi)
444
445        return residual, (g_phi, div_D, rho_ion)
